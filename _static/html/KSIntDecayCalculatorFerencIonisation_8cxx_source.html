70  void KSIntDecayCalculatorFerencIonisation::CalculateLifeTime( const KSParticle& aParticle,
71  double& aLifeTime )
72  {
73  long long tPID = aParticle.GetPID();
74  if ( (tPID == fTargetPID && fTargetPID != 0) || ( (tPID >= fminPID) && ( tPID <= fmaxPID)) )
75  {
76  int n = aParticle.GetMainQuantumNumber();
77  int l = aParticle.GetSecondQuantumNumber();
78 
79  if( n > 150 )
80  {
81  aLifeTime = 1./fCalculator->PBBRionization(fTemperature,n,l,1.,1.e-6,16);
82  }else {
83  aLifeTime = low_n_lifetimes[n-1][l];
84  }
85 
86  } else
87  {
88  aLifeTime = std::numeric_limits<double>::max();
89  }
