430  void InelasticFerencCalculator::randomion( double anE, double& Eloss, double& Theta )
431  {
432  // << "Eloss Computation" << endl;
433 
434  //double Ei=15.45/27.21;
435  double IonizationEnergy_eV = GetIonizationEnergy(); //ionization energy in eV
436  double IonizationEnergy_au = IonizationEnergy_eV / 27.21; //ionization energy in atomic units
437  double c, b, u[ 3 ], K, xmin, ymin, ymax, x, y, T, G, W, Gmax;
438  double q, h, F, Fmin, Fmax, Gp, Elmin, Elmax, qmin, qmax, El, wmax;
439  double WcE, Jstarq, WcstarE, w, D2ion;
440  //int j;
441  double K2, KK, fE, kej, ki, kf, Rex, arg, arctg;
442  //int i;
443  double st1, st2;
444  double Theta_deg;
445  //
446  // I. Generation of Theta
447  // -----------------------
448  Gmax = 1.e-20;
449  if( anE < 200. )
450  Gmax = 2.e-20;
451  T = anE / 27.21;
452  xmin = IonizationEnergy_au * IonizationEnergy_au / (2. * T);
453  b = xmin / (4. * T);
454  ymin = log( xmin );
455  ymax = log( 8. * T + xmin );
456  // Generation of y values with the Neumann acceptance-rejection method:
457  for( int j = 1; j < 5000; j++ )
458  {
459  //subrn(u,2);
460  RandomArray( 3, u );
461  y = ymin + (ymax - ymin) * u[ 1 ];
462  K = exp( y / 2. );
463  c = 1. + b - K * K / (4. * T);
464  G = K * K * (DiffXSecInel( anE, c ) - DiffXSecExc( anE, c ));
465  if( Gmax * u[ 2 ] < G )
466  break;
467  }
468  // y --> x --> c --> Theta
469  x = exp( y );
470  c = 1. - (x - xmin) / (4. * T);
471  Theta = acos( c );
472  Theta_deg = Theta * 180. / KConst::Pi();
473 
474  //
475  // II. Generation of Eloss, for fixed Theta
476  // ----------------------------------------
477  //
478  // For anE<=200 eV we use subr. gensecelen
479  // (in this case no correlation between Theta and Eloss)
480  if( anE <= 200. )
481  {
482  gensecelen( anE, W );
483  Eloss = IonizationEnergy_eV + W;
484  return;
485  }
486  // For Theta>=20 the free electron model is used
487  // (with full correlation between Theta and Eloss)
488  if( Theta_deg >= 20. )
489  {
490  Eloss = anE * (1. - c * c);
491  if( Eloss < IonizationEnergy_eV + 0.05 )
492  Eloss = IonizationEnergy_eV + 0.05;
493  return;
494  }
495  // For anE>200 eV and Theta<20: analytical first Born approximation
496  // formula of Bethe for H atom (with modification for H2)
497  //
498  // Calc. of wmax:
499  if( Theta_deg >= 0.7 )
500  wmax = 1.1;
501  else if( Theta_deg <= 0.7 && Theta_deg > 0.2 )
502  wmax = 2.;
503  else if( Theta_deg <= 0.2 && Theta_deg > 0.05 )
504  wmax = 4.;
505  else
506  wmax = 8.;
507  // We generate the q value according to the Jstarq pdf. We have to
508  // define the qmin and qmax limits for this generation:
509  K = sqrt( 4. * T * (1. - IonizationEnergy_au / (2. * T) - sqrt( 1. - IonizationEnergy_au / T ) * c) );
510  Elmin = IonizationEnergy_au;
511  Elmax = (anE + IonizationEnergy_eV) / 2. / 27.2;
512  qmin = Elmin / K - K / 2.;
513  qmax = Elmax / K - K / 2.;
514  //
515  q = qmax;
516  Fmax = 1. / 2. + 1. / KConst::Pi() * (q / (1. + q * q) + atan( q ));
517  q = qmin;
518  Fmin = 1. / 2. + 1. / KConst::Pi() * (q / (1. + q * q) + atan( q ));
519  h = Fmax - Fmin;
520  // Generation of Eloss with the Neumann acceptance-rejection method:
521  for( int j = 1; j < 5000; j++ )
522  {
523  // Generation of q with inverse transform method
524  // (we use the Newton-Raphson method in order to solve the nonlinear eq.
525  // for the inversion) :
526  RandomArray( 3, u );
527  //subrn(u,2);
528  F = Fmin + h * u[ 1 ];
529  y = 0.;
530  for( int i = 1; i <= 30; i++ )
531  {
532  G = 1. / 2. + (y + sin( 2. * y ) / 2.) / KConst::Pi();
533  Gp = (1. + cos( 2. * y )) / KConst::Pi();
534  y = y - (G - F) / Gp;
535  if( fabs( G - F ) < 1.e-8 )
536  break;
537  }
538  q = tan( y );
539  // We have the q value, so we can define El, and calculate the weight:
540  El = q * K + K * K / 2.;
541  // First Born approximation formula of Bethe for e-H ionization:
542  KK = K;
543  ki = sqrt( 2. * T );
544  kf = sqrt( 2. * (T - El) );
545  K2 = 4. * T * (1. - El / (2. * T) - sqrt( 1. - El / T ) * c);
546  if( K2 < 1.e-9 )
547  K2 = 1.e-9;
548  K = sqrt( K2 ); // momentum transfer
549  Rex = 1. - K * K / (kf * kf) + K2 * K2 / (kf * kf * kf * kf);
550  kej = sqrt( 2. * fabs( El - IonizationEnergy_au ) + 1.e-8 );
551  st1 = K2 - 2. * El + 2.;
552  if( fabs( st1 ) < 1.e-9 )
553  st1 = 1.e-9;
554  arg = 2. * kej / st1;
555  if( arg >= 0. )
556  arctg = atan( arg );
557  else
558  arctg = atan( arg ) + KConst::Pi();
559  st1 = (K + kej) * (K + kej) + 1.;
560  st2 = (K - kej) * (K - kej) + 1.;
561  fE = 1024. * El * (K2 + 2. / 3. * El) / (st1 * st1 * st1 * st2 * st2 * st2) * exp( -2. / kej * arctg ) / (1. - exp( -2. * KConst::Pi() / kej ));
562  D2ion = 2. * kf / ki * Rex / (El * K2) * fE;
563  K = KK;
564  //
565  WcE = D2ion;
566  Jstarq = 16. / (3. * KConst::Pi() * (1. + q * q) * (1. + q * q));
567  WcstarE = 4. / (K * K * K * K * K) * Jstarq;
568  w = WcE / WcstarE;
569  if( wmax * u[ 2 ] < w )
570  break;
571  }
572  //
573  Eloss = El * 27.21;
574  if( Eloss < IonizationEnergy_eV + 0.05 )
575  Eloss = IonizationEnergy_eV + 0.05;
576  //
577  // << "Eloss " << Eloss << endl;
578 
579  return;
580 
581  } //end randomion
