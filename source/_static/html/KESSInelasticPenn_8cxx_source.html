95  KThreeVector tOrthogonalTwo = tInitialDirection.Cross( tOrthogonalOne );
96  KThreeVector tFinalDirection = tInitialDirection.Magnitude() * (sin( tTheta ) * (cos( tPhi ) * tOrthogonalOne.Unit() + sin( tPhi ) * tOrthogonalTwo.Unit()) + cos( tTheta ) * tInitialDirection.Unit());

145  double KESSInelasticPenn::CalculateEnergyLoss( const double& EKin )
146  {
147  if( EKin > 1.0 && EKin < 50000.0 )
148  {
149  double energyLoss = 0.;
150  double resultAbove = 0.;
151  double rand3 = KRandom::GetInstance().Uniform();
152 
153  //map iterator to the maps above and below the kinetic energy of the electron
154  std::map< double, std::vector< std::vector< double > > >::iterator mapAbove, mapBelow;
155 
156  //vector iterator for integral (probability)
157  std::vector< double >::iterator intBelow, intAbove;
158 
159  //value (energy loss)
160  double valBelow, valAbove;
161 
162  //find dictionary entry - mapAbove is the energy area ABOVE kinetic energy of the electron
163  mapAbove = fInElScMap.lower_bound( EKin );
164 
165  //mapBelow the energy area BELOW kinetic energy of the electron
166  mapBelow = mapAbove;
167  mapBelow--;
168 
169  //search for the values above and below the probability integral
170  intAbove = mapAbove->second.at( 1 ).end();
171  intBelow = mapAbove->second.at( 1 ).begin();
172 
173  //the integral doesn't start at 0 all the time. get new random number if no value is found
174  while( rand3 <= *intBelow )
175  {
176 
177  rand3 = KRandom::GetInstance().Uniform();
178 
179  }
180  intAbove = std::lower_bound( intBelow, intAbove, rand3 );
181 
182  intBelow = intAbove;
183  intBelow--;
184 
185  //fill the variables with the according energy losses
186  valAbove = mapAbove->second.at( 0 ).at( intAbove - mapAbove->second.at( 1 ).begin() );
187  valBelow = mapAbove->second.at( 0 ).at( intBelow - mapAbove->second.at( 1 ).begin() );
188 
189  //interpolate to actual random number and save
190  resultAbove = this->InterpolateLinear( rand3, *intBelow, *intAbove, valBelow, valAbove );
191 
192  //see above (but now for the lower energy area)
193  intAbove = mapBelow->second.at( 1 ).end();
194  intBelow = mapBelow->second.at( 1 ).begin();
195 
196  intAbove = lower_bound( intBelow, intAbove, rand3 );
197 
198  intBelow = intAbove;
199  intBelow--;
200 
201  valAbove = mapBelow->second.at( 0 ).at( intAbove - mapBelow->second.at( 1 ).begin() );
202 
203  valBelow = mapBelow->second.at( 0 ).at( intBelow - mapBelow->second.at( 1 ).begin() );
204 
205  double resultBelow = this->InterpolateLinear( rand3, *intBelow, *intAbove, valBelow, valAbove );
206 
207  //interpolate between the results for energy areas above and below the kinetic energy of the electron
208  energyLoss = this->InterpolateLinear( EKin, mapBelow->first, mapAbove->first, resultBelow, resultAbove );
209 
210  if( energyLoss > EKin )
211  {
212  energyLoss = EKin;
213  }
214 
215  if( energyLoss > EKin - energyLoss )
216  {
217  energyLoss = EKin - energyLoss;
218  }
219  return energyLoss;
220  }
221  else
222  {
223  intmsg( eError ) << "Energy of particle out of range for PDF/MFP tables." << ret << " Please check your Terminators to ensure that the particle energy" << " is always between 1 < E < 50,000 eV when using Penn!" << eom;
224  return 0.0;
225  }
