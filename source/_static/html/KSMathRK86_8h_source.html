106  template< class XSystemType >
107  void KSMathRK86< XSystemType >::Integrate(double aTime, const DifferentiatorType& aTerm, const ValueType& anInitialValue, const double& aStep, ValueType& aFinalValue, ErrorType& anError ) const
108  {
109 
110  //do first stage (0) explicitly to deal with possibility of cached data
111 
112  //init value and time
113  fValues[0] = anInitialValue;
114  fIntermediateTime[0] = aTime;
115 
116  //init solution estimates
117  ValueType y6 = fValues[0];
118  ValueType y8 = fValues[0];
119 
120  //we check if we have cached the derivative from the last step
121  if( fHaveCachedDerivative )
122  {
123  fDerivatives[0] = fDerivatives[KSMATHRK86_STAGE];
124  }
125  else
126  {
127  aTerm.Differentiate( fIntermediateTime[0], fValues[0], fDerivatives[0] );
128  }
129 
130  //add contribution to 6th order estimate
131  y6 = y6 + aStep*fB6[0]*fDerivatives[0];
132  //add contribution to 8th order estimate
133  y8 = y8 + aStep*fB8[0]*fDerivatives[0];
134 
135  //compute the value of each stage and
136  //evaluation of the derivative at each stage
137  for(unsigned int i=1; i<KSMATHRK86_STAGE; i++)
138  {
139  //compute the time of this stage
140  fIntermediateTime[i] = fIntermediateTime[0] + aStep*fC[i];
141 
142  //now compute the stage value
143  fValues[i] = fValues[0];
144  for(unsigned int j=0; j<fAColumnLimit[i]; j++)
145  {
146  fValues[i] = fValues[i] + (aStep*fA[i][j])*fDerivatives[j];
147  }
148 
149  //now compute the derivative term for this stage
150  aTerm.Differentiate(fIntermediateTime[i], fValues[i], fDerivatives[i]);
151 
152  //add contribution to 6th order estimate
153  y6 = y6 + aStep*fB6[i]*fDerivatives[i];
154 
155  //add contribution to 8th order estimate
156  y8 = y8 + aStep*fB8[i]*fDerivatives[i];
157  }
158 
159  //we use the 8th order estimate for the solution (local extrapolation)
160  aFinalValue = y8;
161 
162  //now estimate the truncation error on the step (for stepsize control)
163  anError = (y6 - y8);
164 
165  //evaluate the derivative at final point and cache it for the next
166  //step (this derivative is needed for the dense output interpolation)
167  fIntermediateTime[KSMATHRK86_STAGE] = aTime + aStep;
168  aTerm.Differentiate( fIntermediateTime[KSMATHRK86_STAGE], aFinalValue, fDerivatives[KSMATHRK86_STAGE] );
169  fHaveCachedDerivative = true;
170 
171 
172  return;
173  }
